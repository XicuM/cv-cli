% Custom commands

% Custom blue bullet
\newcommand{\bluebullet}{%
    \hspace{0.09cm}%
    \tikz[baseline=-0.14cm]{%
        \fill[white] (0,0) circle (0.16cm);%
        \fill[primary] (0,0) circle (0.08cm);%
    }%
}

% Custom hollow bullet
\newcommand{\hollowbullet}{%
    \makebox[0pt][l]{%
        \hspace{-0.4cm}%
        \raisebox{0.04cm}{%
            \tikz[baseline=(char.base)]{%
                \draw[secondary] (0,0) circle (0.06cm);%
            }%
        }%
    }%
}

% Date command
\newcommand{\dates}[2]{%
    \hfill\raisebox{0.04cm}{\footnotesize\color{primary}\textbf{#1 - #2}}%
}

% Reusable boxes to measure wrapped title height
\newsavebox{\entrytitlebox}
\newsavebox{\subentrytitlebox}

% Entry header with wrapped title and fixed date column
\newcommand{\entryheader}[3]{%
    \sbox{\entrytitlebox}{\parbox[t]{0.60\linewidth}{\raggedright\textsc{#1}}}%
    \noindent\makebox[0pt][l]{\bluebullet\hspace{0.20cm}}%
    \hspace{0.65cm}%
    \begin{minipage}[t]{0.60\linewidth}%
        \usebox{\entrytitlebox}%
    \end{minipage}%
    \hfill%
    \hspace{0.06cm}%
    \begin{minipage}[t]{0.27\linewidth}%
        \raggedleft\raisebox{0.04cm}{\footnotesize\color{primary}\textbf{#2 - #3}}%
    \end{minipage}\par%
    \ifdim\dimexpr\ht\entrytitlebox+\dp\entrytitlebox\relax>1.6\baselineskip
        \vspace{0.85ex}%
    \else
        \vspace{0.18ex}%
    \fi%
}

% Sub-entry header with wrapped title and fixed date column
\newcommand{\subentryheader}[3]{%
    \sbox{\subentrytitlebox}{\parbox[t]{0.56\linewidth}{\raggedright\textsc{#1}}}%
    \noindent\hspace{1cm}%
    \begin{minipage}[t]{0.56\linewidth}%
        \usebox{\subentrytitlebox}%
    \end{minipage}%
    \hfill%
    \hspace{0.10cm}%
    \begin{minipage}[t]{0.27\linewidth}%
        \raggedleft\raisebox{0.04cm}{\footnotesize\color{primary}\textbf{#2 - #3}}%
    \end{minipage}\par%
    \ifdim\dimexpr\ht\subentrytitlebox+\dp\subentrytitlebox\relax>1.6\baselineskip
        \vspace{0.75ex}%
    \else
        \vspace{0.18ex}%
    \fi%
}

% Section header for left column
\newcommand{\headleft}[1]{%
    \vspace*{0.4cm}\textsc{\textbf{#1}}\par%
    \vspace*{-1.1ex}\hrulefill\par\vspace*{0.12cm}%
}

% Tighter first section header (used below the name block)
\newcommand{\headleftfirst}[1]{%
    \vspace*{0.16cm}\textsc{\textbf{#1}}\par%
    \vspace*{-1.1ex}\hrulefill\par\vspace*{0.12cm}%
}

% Section header for right column
\newcommand{\headright}[1]{%
    \textsc{\large\color{primary}#1}\par%
    \vspace*{-0.25cm}{\color{primary}\hrulefill}\par%
    \vspace*{0.25cm}%
}

% Counter for unique section identifiers
\newcounter{sectioncounter}

% Counter for unique subposition identifiers
\newcounter{subposcounter}

% Section with vertical line
\newcommand{\newsection}[2]{%
    \stepcounter{sectioncounter}%
    \headright{#1}%
    \tikzmark{start-\arabic{sectioncounter}}%
    #2%
    \par\noindent%
    \tikzmark{end-\arabic{sectioncounter}}%
    \begin{tikzpicture}[remember picture, overlay]%
        \coordinate (line-start-\arabic{sectioncounter}) at ([xshift=0.25cm,yshift=0cm]pic cs:start-\arabic{sectioncounter});%
        \coordinate (line-end-\arabic{sectioncounter}) at ([xshift=0.25cm,yshift=0.75cm]pic cs:end-\arabic{sectioncounter});%
    \end{tikzpicture}%
}

% Detail item command
\newcommand{\detail}[1]{%
    {\small\color{secondary}#1}%
}

% Company/location line for entries and sub-entries
\newcommand{\companyline}[3]{%
    \hspace{#1}#2#3 \\[0.1cm]%
}

% Reusable details list with configurable left margin
\newcommand{\detailslist}[2][1.12cm]{%
    \begin{itemize}[leftmargin=#1]%
        #2%
    \end{itemize}%
}

% Draw vertical line for a sub-position timeline
\newcommand{\drawsubpositionline}[2]{%
    \begin{tikzpicture}[remember picture, overlay]%
        \coordinate (line-start-subpos-\arabic{subposcounter}) at ([xshift=0.70cm, yshift=#1]pic cs:start-subpos-\arabic{subposcounter});%
        \coordinate (line-end-subpos-\arabic{subposcounter}) at ([xshift=0.70cm, yshift=#2]pic cs:end-subpos-\arabic{subposcounter});%
        \draw[primary!30, line width=1.8pt] (line-start-subpos-\arabic{subposcounter}) -- (line-end-subpos-\arabic{subposcounter});%
    \end{tikzpicture}%
}

% Reusable subposition block
\newcommand{\subpositionentry}[7]{%
    \stepcounter{subposcounter}%
    \tikzmark{start-subpos-\arabic{subposcounter}}%
    \subentryheader{#1}{#2}{#3}%
    #4%
    \detailslist[1.5cm]{#5}%
    \tikzmark{end-subpos-\arabic{subposcounter}}%
    \par\vspace{-0.25cm}%
    \drawsubpositionline{#6}{#7}%
}

% Badge command
\newcommand{\skillbadge}[1]{%
    \tikz[baseline=(char.base)]{%
        \node[%
            shape=rectangle, %
            rounded corners=3pt, %
            inner sep=2pt, %
            fill=primary!10%
        ] (char) {\small\textbf{#1}};%
    }%
}

% Skill category label
\newcommand{\skillcategory}[1]{%
    {\small\textit{#1}}\\%
}

% Circular image command with diffused shadow
\newcommand{\circularimage}[2][2.5cm]{%
    \begin{tikzpicture}%
        % Diffused shadow
        \fill[black!15] (0.08,-0.08) circle (#1/2 + 0.3mm);%
        \fill[black!20] (0.06,-0.06) circle (#1/2 + 0.2mm);%
        \fill[black!25] (0.04,-0.04) circle (#1/2 + 0.1mm);%
        \fill[black!30] (0.02,-0.02) circle (#1/2);%
        % Main circular image
        \begin{scope}%
            \clip (0,0) circle (#1/2);%
            \node[anchor=center] at (0,0) {\includegraphics[width=#1]{#2}};%
        \end{scope}%
    \end{tikzpicture}%
}
